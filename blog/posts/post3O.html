Why LeetCode

Outside of the things I've mentioned, I've discovered the allure of LeetCode in the past few weeks. Initially, I thought of it as nothing more than a ProjectEuler copycat, but it differs from PE in the sense that it is a code-first problem site. With Project Euler, the goal was to generate the answer, regardless of language or the amount of time it takes (although they do note that solutions over a minute are not intended). This means understanding the complexity of algorithms and data structures isn't all that important. Rather, the focus is on the problem solving that involves taking a challenging problem and digesting it into a form that can be computed via code.

On the other hand, LeetCode is almost entirely about data structure and algorithm optimization. The problems are significantly easier than those of Project Euler, but the constraints are a lot tighter, too, and that tests a very different skill set.

Needless to say, I was a bit shell-shocked when I entered my first contest (before I had even done a single LeetCode problem). Not only was I in the bottom 50%, but I wasn't even able to find a solution to the last problem, let alone an efficient one. That motivated me to get better at LeetCode, so I found a list of standard LeetCode problems to solve, and got started immediately. And I finally finished!

These solutions aren't necessarily the most optimal, but they all were entirely my own creation, and I'm pretty proud of how far I've come!

I'm only going to post some of the medium and hard problems since most of the easy ones are trivial.

Product of Array Except Self 
from itertools import accumulate
def productExceptSelf(self, nums: List[int]) -> List[int]:
    output = list(accumulate([1]+nums[:-1],lambda x,y:x*y))
    accumulate_right = nums[-1]
    for i in range(len(output)-2,-1,-1):
        output[i] *= accumulate_right
        accumulate_right *= nums[i]
    return output

Maximum Product Subarray 
def maxProduct(self, nums: List[int]) -> int:
    if len(nums) == 1:
        return nums[0]
    max_,curr_max,curr_min = [0]*3
    for n in nums:
        if n==0:
            max_ = max(max_,curr_max)
            curr_max,curr_min = 0,0
        elif n > 0:
            curr_max = max(n,curr_max*n)
            curr_min = min(n,curr_min*n)
        else:
            curr_max = max(n,curr_min*n)
            curr_min = min(n,curr_max*n)
    return max(max_,curr_max)

Find Minimum in Rotated Sorted Array 
def findMin(self, nums: List[int]) -> int:
    lo = 0
    hi = len(nums)
    if nums[hi-1]>=nums[lo]:
        return nums[lo]
    while True:
        mid = (lo + hi)//2
        if nums[mid] > nums[hi-1]:
            lo = mid + 1
        elif nums[mid-1] > nums[mid]:
            return nums[mid]
        else:
            hi = mid

Search in Rotated Sorted Array 
def search(self, nums: List[int], target: int) -> int:
    lo = 0
    hi = len(nums)
    while hi > lo:
        mid = (lo+hi)//2
        if nums[mid] == target:
            return mid
        if nums[mid] > target:
            if nums[mid] <= nums[hi-1]:
                hi = mid
            else:
                lo = mid + 1
        else:
            if nums[mid] >= nums[lo]:
                lo = mid + 1
            else:
                hi = mid
    return -1

3 Sum 
def threeSum(self, nums: List[int]) -> List[List[int]]:
    target = -10**5-1
    nums.sort()
    output = []
    length = len(nums)
    i = 0
    while True:
        while length > i and target == -nums[i]:
            i += 1
        if i == length:
            return output
        target = -nums[i]
        l = i + 1
        h = length - 1
        while h > l:
            if nums[l]+nums[h] == target:
                output.append([nums[i],nums[l],nums[h]])
                old_l = l
                while h > l and nums[old_l] == nums[l]:
                    l += 1
                old_h = h
                while h > l and nums[old_h] == nums[h]:
                    h -= 1
            elif target > nums[l]+nums[h]:
                old_l = l
                while h > l and nums[old_l] == nums[l]:
                    l += 1
            else:
                old_h = h
                while h > l and nums[old_h] == nums[h]:
                    h -= 1

Counting Bits 
def countBits(self, n: int) -> List[int]:
    l = [0]
    for i in range(1,n+1):
        l.append(l[i>>1]+i%2)
    return l

Coin Change 
def coinChange(self, coins: List[int], amount: int) -> int:
    dp = [0]
    for i in range(1,amount+1):
        dp.append(min((dp[i-c]+1 for c in coins if i>=c),default=amount+1))
    return dp[-1] if dp[-1]<=amount else -1

Longest Increasing Subsequence 
def lengthOfLIS(self, nums: List[int]) -> int:
    l = []
    for n in nums:
        k = bisect(l,n)
        if k and n == l[k-1]:
            continue
        if len(l) > k:
            l[k] = n
            continue
        l.append(n)
    return len(l)

Longest Common Subsequence 
def longestCommonSubsequence(self, text1: str, text2: str) -> int:
    n,m = len(text1),len(text2)
    l = [[0]*(n+1) for _ in range(2)]
    for i in range(m):
        for j in range(1,n+1):
            if text2[i] == text1[j-1]:
                l[i%2][j] = l[1-i%2][j-1] + 1
            else:
                l[i%2][j] = max(l[1-i%2][j],l[i%2][j-1])
    return l[1-m%2][n]

Merge k sorted lists 
def output_val(l):
    try:
        return l[1].val
    except:
        return 10001
def mergeKLists(self, lists: List[ListNode]) -> ListNode:
    if not lists:
        return None
    start = min(enumerate(lists),key=output_val)
    if not start[1]:
        return None
    prev = start
    lists[prev[0]] = prev[1].next
    curr = min(enumerate(lists),key=output_val)
    while curr[1]:
        prev[1].next = curr[1]
        prev = curr
        lists[prev[0]] = prev[1].next
        curr = min(enumerate(lists),key=output_val)
    prev[1].next = None
    return start[1]
